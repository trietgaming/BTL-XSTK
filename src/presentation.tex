\begin{frame}{Tại sao dùng mô hình Log-Linear?}
  \begin{block}{Vấn đề với mô hình Linear}
    \footnotesize
    \begin{itemize}
      \item Giá GPU phân phối \textbf{lệch phải} mạnh (flagship tạo đuôi dài)
      \item Phương sai phần dư \textbf{không đồng nhất} (Heteroscedasticity)
      \item OLS giả định cộng gộp tuyệt đối --- không phù hợp
    \end{itemize}
  \end{block}
  \begin{exampleblock}{Tại sao Log-Linear?}
    \footnotesize Thông số kỹ thuật tác động lên giá theo \textbf{tỷ lệ \%}:
    \vspace{0.1em}

    \[\log(Y) = \beta_0 + \sum \beta_i X_i + \varepsilon\]

    $\beta_i$ đo \% thay đổi giá khi $X_i$ tăng 1 đơn vị. Log nén đuôi, ổn định phương sai.
  \end{exampleblock}
\end{frame}

\begin{frame}{Các kiểm định thống kê sử dụng}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{block}{Welch's ANOVA}
        \scriptsize
        \textbf{Ý nghĩa:} So sánh giá trị trung bình hai nhóm độc lập.\\
        \textbf{Mục tiêu:} Kiểm định chênh lệch giá hãng NVIDIA--AMD khi phương sai không đồng nhất.\\
        $H_0$: $\mu_{\text{NVIDIA}} = \mu_{\text{AMD}}$ (Welch hiệu chỉnh bậc tự do).
      \end{block}
      \vspace{0.4em}
      \begin{block}{VIF --- Đa cộng tuyến}
        \scriptsize
        \textbf{Ý nghĩa:} Đo lường mức độ tương quan giữa các biến độc lập.\\
        \textbf{Mục tiêu:} Phát hiện sự trùng lặp thông tin gây sai lệch hệ số OLS.\\
        $\text{VIF}_j = \dfrac{1}{1 - R^2_j}$ (Ngưỡng $< 10$, $R^2_j$: hồi quy $X_j$ lên các biến khác).
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{Breusch-Pagan}
        \scriptsize
        \textbf{Ý nghĩa:} Kiểm định phương sai sai số thay đổi (Heteroscedasticity).\\
        \textbf{Mục tiêu:} Đảm bảo tính hiệu quả tuyến tính tốt nhất (BLUE) của OLS.\\
        $H_0$: Var$(\varepsilon_i)$ không phụ thuộc $X$ ($p > 0.05 \Rightarrow$ phương sai đồng nhất).
      \end{block}
      \vspace{0.4em}
      \begin{block}{Shapiro-Wilk}
        \scriptsize
        \textbf{Ý nghĩa:} Kiểm định giả định phần dư có phân phối chuẩn.\\
        \textbf{Mục tiêu:} Đảm bảo tính hợp lệ cho các kiểm định hệ số t-test, F-test.\\
        $H_0$: phần dư $\sim \mathcal{N}(0, \sigma^2)$ (mẫu lớn kết hợp Q-Q Plot và CLT).
      \end{block}
  \end{columns}
\end{frame}

clean_data <- selected_data %>%
  mutate(across(where(is.character),

            NA, str_trim(.)))) %>%
  filter(Manufacturer != "Intel") %>%
  mutate(Manufacturer =
    str_replace(Manufacturer,"ATI","AMD")) %>%
  filter(!str_detect(Name, regex(

    ignore_case=TRUE))) %>%
  mutate(Release_Year = as.integer(
    format(as.Date(Release_Date,"%d-%b-%Y"),"%Y"))) %>%
  select(-Release_Date) %>%
  mutate(
    Release_Price = as.numeric(
      str_extract(Release_Price,"\\d+\\.?\\d*")),
    # ... tuong tu Max_Power, Memory, Memory_Bandwidth, Core_Speed
    Manufacturer = as.factor(Manufacturer),
    Memory_Type  = as.factor(Memory_Type)
  ) %>% filter(!is.na(Release_Price))

      \begin{block}{Điểm kỹ thuật}

# IQR Outlier Detection
calc_tukey <- function(vec) {
  v <- na.omit(vec)
  q1 <- quantile(v,0.25); q3 <- quantile(v,0.75)
  iqr <- q3 - q1
  list(lower=q1-1.5*iqr, upper=q3+1.5*iqr)
}

# Trich VRAM bang Regex tu ten GPU
extract_vram <- function(df) {
  df %>% mutate(
    mem_str = str_extract(Name, regex(
      "\\b(\\d+)\\s*(GB|G|MB)\\b",
      ignore_case=TRUE)),
    mem_val = as.numeric(
      str_extract(mem_str,"\\d+")),
    mem_val = ifelse(
      str_detect(mem_str,
        regex("GB|G\\b",ignore_case=TRUE)),
      mem_val*1024, mem_val),
    Memory = ifelse(is.na(Memory)
      & !is.na(mem_val), mem_val, Memory)
  ) %>% select(-mem_str,-mem_val)
}
