\begin{minted}{R}
library(stringr); library(dplyr); library(car); library(lmtest)

# Đọc dữ liệu thô: 3.406 quan sát x 34 biến

\subsection{Bước 2: Chọn lọc, đổi tên và ép kiểu dữ liệu}
\label{sec:buoc_2_chon_loc}

Với 8 biến đã xác định ở Bước~\ref{sec:buoc_1_tai_du_lieu}, Bước~\ref{sec:buoc_2_chon_loc} thực hiện 4 thao tác độc lập và có thứ tự: chọn và đổi tên cột, gộp nhãn ATI $\to$ AMD, trích xuất số từ chuỗi, và parse ngày tháng thành năm.

\subsubsection*{2.1. Chọn cột và đổi tên}

Từ 34 cột ban đầu, chỉ giữ lại đúng 8 cột đã xác định và đổi tên theo quy ước snake\_case để đồng bộ trong suốt pipeline phân tích:

\begin{minted}{R}
df <- raw_data %>%
  select(Release_Price, Max_Power, Memory, Memory_Bus,
         Core_Speed, Release_Date, Manufacturer, Memory_Type) %>%
  rename(
    release_price = Release_Price, tdp          = Max_Power,
    memory_size   = Memory,        memory_bus   = Memory_Bus,
    core_speed    = Core_Speed,    release_date = Release_Date,
    manufacturer  = Manufacturer,  memory_type  = Memory_Type
  )
\end{minted}

\textbf{\texttt{select()}}: Thu hẹp data frame từ 34 xuống 8 cột. Ở thời điểm này tất cả biến vẫn ở kiểu \texttt{chr} (chuỗi ký tự) --- chưa có bất kỳ ép kiểu nào. \textbf{\texttt{rename()}}: Đổi tên sang snake\_case ngắn gọn để code ở các bước sau dễ đọc và ít lỗi chính tả hơn.

\subsubsection*{2.2. Gộp nhãn ATI $\to$ AMD}

Lý do gộp đã trình bày tại Phần~2.2; đây là bước thực thi:

\begin{minted}{R}
df <- df %>%
  mutate(
    manufacturer = ifelse(str_trim(manufacturer) == "ATI", "AMD",
                          str_trim(manufacturer))
  )
\end{minted}

\textbf{\texttt{str\_trim()}}: Xóa khoảng trắng đầu/cuối chuỗi --- CSV từ web scraper thường có ký tự thừa vô hình. \textbf{\texttt{ifelse()}}: Thay thế trực tiếp chuỗi \texttt{"ATI"} bằng \texttt{"AMD"}; các nhãn khác (\texttt{"NVIDIA"}, \texttt{"Intel"}) giữ nguyên.

Kết quả kiểm tra sau khi gộp:
\begin{rconsole}
> table(df$manufacturer)
  AMD Intel NVIDIA

\end{rconsole}

Toàn bộ card ATI đã được gán vào nhóm AMD; Intel (4 quan sát) sẽ bị loại bỏ ở Bước~\ref{sec:buoc_4_dien_khuyết} vì không có giá niêm yết.

\subsubsection*{2.3. Trích xuất số từ chuỗi có đơn vị}

Các cột số trong CSV gốc được lưu kèm đơn vị dưới dạng chuỗi ký tự (ví dụ: \texttt{"738 MHz"}, \texttt{"141 Watts"}, \texttt{"4096 MB"}). Cần trích xuất phần số và chuyển sang kiểu \texttt{numeric}:

\begin{minted}{R}
df <- df %>%
  mutate(
    release_price = as.numeric(str_extract(release_price, "\\d+\\.?\\d*")),
    tdp           = as.numeric(str_extract(tdp,           "\\d+\\.?\\d*")),
    memory_size   = as.numeric(str_extract(memory_size,   "\\d+\\.?\\d*")),
    memory_bus    = as.numeric(str_extract(memory_bus,    "\\d+\\.?\\d*")),
    core_speed    = as.numeric(str_extract(core_speed,    "\\d+\\.?\\d*"))
  )
\end{minted}

Regex \texttt{"{\textbackslash}d+{\textbackslash}.?{\textbackslash}d*"} khớp với: một hoặc nhiều chữ số, theo sau là dấu chấm thập phân tùy chọn và các chữ số thập phân --- ví dụ: \texttt{"738 MHz"} $\to$ \texttt{"738"} $\to$ \texttt{738}; \texttt{"14.999"} $\to$ \texttt{"14.999"} $\to$ \texttt{14999}.

Chuỗi không hợp lệ (\texttt{""}, \texttt{"-"}, \texttt{"N/A"}) không khớp regex $\to$ \texttt{str\_extract()} trả về \texttt{NA} $\to$ \texttt{as.numeric(NA)} trả về \texttt{NA}. Đây là hành vi \textit{mong muốn}: biến các chuỗi rỗng thành \texttt{NA} thực sự để bước điền khuyết ở Bước~\ref{sec:buoc_4_dien_khuyết} xử lý nhất quán.

\subsubsection*{2.4. Parse chuỗi ngày tháng $\to$ năm}

Cột \texttt{release\_date} lưu dưới định dạng \texttt{"01-Mar-2009"} --- không thể dùng trực tiếp trong hồi quy. Cần trích xuất số nguyên là năm:

\begin{minted}{R}
df <- df %>%
  mutate(
    release_year = as.integer(
      format(as.Date(str_trim(release_date), "%d-%b-%Y"), "%Y"))
  ) %>%
  select(-release_date)
\end{minted}

\textbf{\texttt{as.Date(..., "\%d-\%b-\%Y")}}: Parse chuỗi theo định dạng ngày-tháng viết tắt-năm (\texttt{"\%b"} là Jan, Feb, Mar\ldots). \textbf{\texttt{format(..., "\%Y")}}: Trích lấy 4 chữ số năm dưới dạng chuỗi. \textbf{\texttt{as.integer()}}: Chuyển thành số nguyên liên tục phù hợp cho hồi quy --- hệ số $\hat{\beta}$ đọc được là ``mức thay đổi log-giá mỗi năm''. \textbf{\texttt{select(-release\_date)}}: Bỏ cột chuỗi gốc sau khi đã tách năm, tránh nhầm lẫn ở các bước sau.

Chuỗi không parse được (thiếu ngày, định dạng sai) tự động trở thành \texttt{NA} và được xử lý ở Bước~\ref{sec:buoc_4_dien_khuyết}.

\subsection{Bước 3: Nhận diện ngoại lai (IQR trên tập gốc)}
\label{sec:buoc_3_ngoai_lai}

Theo nguyên tắc đã trình bày ở Section~\ref{sec:co_so_ly_thuyet} (Phần 2), ranh giới IQR phải được tính \textbf{trước khi lọc hay điền khuyết}. Nhóm tính ngưỡng Tukey trên tập thô rồi đưa trực tiếp vào bộ lọc Bước~\ref{sec:buoc_4_dien_khuyết}:

\begin{minted}{R}
# Tính ngưỡng Tukey cho release_price trên tập thô (N = 3.406)
price_raw_vec <- na.omit(df$release_price)
price_q3_raw  <- quantile(price_raw_vec, 0.75)
price_iqr_raw <- price_q3_raw - quantile(price_raw_vec, 0.25)
price_upper   <- price_q3_raw + 1.5 * price_iqr_raw  # => $814

# Hàm đếm giá trị ngoại lai còn lại (dùng cho TDP, core_speed)
count_outliers_by_raw_bounds <- function(raw_vec, clean_vec) {
  raw_vec <- na.omit(raw_vec)
  q3 <- quantile(raw_vec, 0.75); iqr <- q3 - quantile(raw_vec, 0.25)
  return(sum(clean_vec > q3 + 1.5 * iqr, na.rm = TRUE))
}
\end{minted}

Ngưỡng trên của \texttt{release\_price} tính được là \$814. Toàn bộ 33 quan sát vượt ngưỡng này đều thuộc phân khúc Workstation/Enterprise (lý do loại bỏ tại Section~\ref{sec:co_so_ly_thuyet}). Kết quả được trực quan hóa qua Hình~\ref{fig:boxplot_outliers}.

\subsection{Bước 4: Loại quan sát không hợp lệ và điền giá trị khuyết}
\label{sec:buoc_4_dien_khuyết}

Bước này thực hiện hai thao tác tuần tự: (i)~lọc bỏ các dòng thiếu biến phụ thuộc và hãng Intel, (ii)~điền khuyết cho các biến còn lại theo chiến lược đã trình bày tại Bảng~\ref{tab:imputation_strategy} (Phần~2).

\subsubsection*{4.1. Lọc bỏ quan sát không hợp lệ}

\begin{minted}{R}
df_clean <- df %>%
  filter(!is.na(release_price)) %>%        # Giữ dòng có giá (biến Y)
  filter(manufacturer != "Intel") %>%       # Loại Intel (không có giá)
  filter(release_price <= price_upper)      # Loại GPU workstation/enterprise
\end{minted}

Loại bỏ 2.846 dòng thiếu \texttt{release\_price}, 4 dòng Intel, và 33 GPU Workstation/Enterprise có giá $> \$814$ (ngưỡng Tukey từ Bước~\ref{sec:buoc_3_ngoai_lai}). Tập phân tích còn \textbf{523 quan sát}.

\subsubsection*{4.2. Điền khuyết theo vai trò biến}

\begin{minted}{R}
df_clean <- df_clean %>%
  mutate(
    # Biến số liên tục: điền Median
    tdp = ifelse(is.na(tdp), median(tdp, na.rm = TRUE), tdp),
    core_speed = ifelse(is.na(core_speed),
                        median(core_speed, na.rm = TRUE), core_speed),
    release_year = ifelse(is.na(release_year),
                          median(release_year, na.rm = TRUE), release_year),
    # Biến số rời rạc: điền Mode
    memory_size = ifelse(is.na(memory_size),
                         get_mode(memory_size), memory_size),
    memory_bus  = ifelse(is.na(memory_bus),
                         get_mode(memory_bus),  memory_bus),
    # Biến phân loại: điền Mode
    memory_type  = ifelse(is.na(memory_type) | memory_type == "",
                          get_mode(memory_type), memory_type),
    manufacturer = ifelse(is.na(manufacturer) | manufacturer == "",
                          get_mode(manufacturer), manufacturer)
  )
\end{minted}

Số lượng NA được điền trên 523 quan sát: \texttt{tdp}, \texttt{core\_speed}, \texttt{release\_year} (Median); \texttt{memory\_size}, \texttt{memory\_bus}, \texttt{memory\_type} và \texttt{manufacturer} (Mode) --- theo chiến lược Bảng~\ref{tab:imputation_strategy}.

Sau quy trình, \texttt{colSums(is.na(df\_clean))} xác nhận toàn bộ 8 biến đều có 0~NA --- bộ dữ liệu hoàn toàn sạch, sẵn sàng cho bước mã hóa và chuẩn hóa.

\textbf{7.1.2 Mô hình cơ sở (Y gốc) và kiểm định giả định OLS}

Sau khi loại \texttt{memory\_type}, nhóm xây mô hình cơ sở với biến phụ thuộc là giá gốc để kiểm định các giả định OLS trước khi quyết định có cần biến đổi hay không:

\begin{minted}{R}
mlr_model_base <- lm(release_price ~ tdp + memory_size + memory_bus +
                       core_speed + manufacturer + release_year, data = df_final)

print(vif(mlr_model_base))       # Xác nhận VIF < 5
bp_base <- bptest(mlr_model_base)
sw_base <- shapiro.test(residuals(mlr_model_base))
print(bp_base); print(sw_base)
\end{minted}

Kết quả: $BP = 63{,}43$, $p < 2{,}2 \times 10^{-16}$ --- phương sai sai số không đồng nhất; $W = 0{,}942$, $p < 10^{-12}$ --- phần dư chưa hoàn toàn chuẩn. Cả hai giả định OLS đều vi phạm $\Rightarrow$ áp dụng log-transform lên $Y$ để cải thiện.

\textbf{7.1.3 Mô hình tinh chỉnh Log-Linear}

\begin{minted}{R}
log_model_final <- lm(log(release_price) ~ tdp + memory_size + memory_bus +
                          core_speed + manufacturer + release_year, data = df_final)

summary(log_model_final)
\end{minted}

Log-transform cải thiện ổn định phương sai và đưa phần dư gần chuẩn hơn. Kết quả \texttt{summary()} cho thấy tất cả 6 biến đều có ý nghĩa thống kê ($p < 0{,}001$) với $R^2 = 75{,}78\%$. Bảng hệ số chi tiết được trình bày tại Phần~4.

\textbf{7.1.4 Kiểm định lại giả định trên mô hình Log-Linear}

\begin{minted}{R}
print(vif(log_model_final))
print(bptest(log_model_final))
print(shapiro.test(residuals(log_model_final)))
\end{minted}

\begin{sloppypar}
\textbf{Breusch-Pagan}: $BP = 58{,}58$, $p < 2{,}2 \times 10^{-16}$ --- phương sai vẫn chưa hoàn toàn đồng nhất nhưng cải thiện so với mô hình gốc ($BP = 63{,}43$). \textbf{Shapiro-Wilk}: $W = 0{,}946$, $p = 7{,}537 \times 10^{-13}$ --- tương đương mô hình gốc ($W = 0{,}942$). Mô hình log-linear được chọn làm mô hình chính thức vì $R^2$ cao hơn ($75{,}78\%$ vs $72{,}73\%$) và hệ số có ý nghĩa kinh tế trực tiếp; diễn giải chi tiết tại Phần~4.
\end{sloppypar}

Giá dự đoán được khôi phục về đơn vị USD bằng \texttt{exp(predict(log\_model\_final))} và so sánh trực quan với giá thực tế qua Hình~\ref{fig:scatter_pred} (Phần~4).



\subsubsection{7.2. Phân tích phương sai}

\begin{minted}{R}
# Kiểm định đồng nhất phương sai
print(leveneTest(release_price ~ manufacturer, data = df_final))

# Welch's ANOVA (var.equal = FALSE để xử lý phương sai không đồng nhất)
anova_result <- oneway.test(release_price ~ manufacturer,
                            data = df_final, var.equal = FALSE)
print(anova_result)
\end{minted}

\textbf{Levene's Test} kiểm tra tính đồng nhất phương sai giữa hai nhóm. \textbf{oneway.test(..., var.equal = FALSE)} thực hiện Welch's ANOVA --- phiên bản bền vững khi phương sai không đồng nhất. Kết quả chi tiết được trình bày tại Phần~4.
